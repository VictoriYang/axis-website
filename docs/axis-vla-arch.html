# Axis VLA Forward Pass class AxisVLA(nn.Module): def forward(self, image, language, state): # 1. 编码 vision_tokens = self.vit_encoder(image) lang_tokens = self.llm_encoder(language) action_tokens = self.state_encoder(state) # 2. 统一潜在空间融合 fused_tokens = self.moe_layer( vision_tokens, lang_tokens, action_tokens ) # 3. 三专家处理 scene_feat = self.scene_expert(fused_tokens) world_feat = self.world_expert(fused_tokens) action_feat = self.action_expert(fused_tokens) # 4. 统一注意力交互 output = self.unified_attention( scene_feat, world_feat, action_feat ) # 5. Diffusion 输出动作 action = self.diffusion_head(output) # 6. (可选) 世界模型预测 future_frame = self.world_model(output) return action, future_frame
